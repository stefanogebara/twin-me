export const getPlatformLogo = (platformKey: string | undefined): React.FC<{ className?: string }> | null => {
  if (!platformKey) return null;
  return PlatformLogos[platformKey.toLowerCase()] || null;
};
 
// Generic PlatformLogo component that renders the appropriate logo based on platform/provider prop
export const PlatformLogo: React.FC<{
  platform?: string;
  provider?: string;
  className?: string;
  size?: number;
}> = ({ platform, provider, className, size }) => {
  // Support both 'platform' and 'provider' prop names for backward compatibility
  const platformKey = platform || provider;
 
  // Handle size prop by using inline styles or className
  const finalClassName = className || 'w-8 h-8';
  const sizeStyle = size ? { width: `${size}px`, height: `${size}px` } : undefined;
 
  const LogoComponent = getPlatformLogo(platformKey);
 
  if (LogoComponent) {
    return (
      <div style={sizeStyle} className={sizeStyle ? undefined : finalClassName}>
        <LogoComponent className={sizeStyle ? 'w-full h-full' : finalClassName} />
      </div>
    );
  }
 
  // No logo found - return null instead of showing placeholder
  return null;
};
