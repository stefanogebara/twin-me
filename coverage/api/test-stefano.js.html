import dotenv from 'dotenv';
dotenv.config();
 
const apiKey = process.env.SCRAPIN_API_KEY;
const linkedInUrl = 'https://www.linkedin.com/in/stefano-gebara-9b2a39197/';
 
console.log('Testing Scrapin for Stefano...');
 
const params = new URLSearchParams({
  apikey: apiKey,
  linkedInUrl: linkedInUrl
});
 
const response = await fetch('https://api.scrapin.io/v1/enrichment/profile?' + params);
const result = await response.json();
 
if (!result.success) {
  console.log('Error:', result);
  process.exit(1);
}
 
const person = result.person;
console.log('\n=== PROFILE ===');
console.log('Name:', person.firstName, person.lastName);
console.log('Headline:', person.headline);
console.log('Location:', JSON.stringify(person.location));
console.log('Connections:', person.connectionsCount);
console.log('Followers:', person.followerCount);
 
const positions = person.positions?.positionHistory || [];
console.log('\n=== POSITIONS (' + positions.length + ') ===');
positions.forEach((pos, i) => {
  console.log('\n[' + (i+1) + '] ' + pos.title + ' at ' + pos.companyName);
  console.log('    Dates:', JSON.stringify(pos.startEndDate));
  console.log('    Location:', pos.location || '(none)');
  console.log('    Description:', (pos.description || '(none)').substring(0, 300));
});
 
const schools = person.schools?.educationHistory || [];
console.log('\n=== EDUCATION (' + schools.length + ') ===');
schools.forEach((s, i) => {
  console.log('[' + (i+1) + '] ' + s.schoolName + ' - ' + (s.degreeName || '') + ' ' + (s.fieldOfStudy || ''));
  console.log('    Years:', s.startEndDate?.start?.year, '-', s.startEndDate?.end?.year);
});
 
console.log('\n=== SKILLS ===');
console.log((person.skills || []).slice(0, 15).join(', '));
 
console.log('\n=== SUMMARY ===');
console.log(person.summary || '(none)');
 
console.log('\n=== RAW PERSON OBJECT ===');
console.log(JSON.stringify(person, null, 2));
